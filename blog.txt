  var currentCategory = null;
  $.each(images, function (index, image) {
    if (image.category !== currentCategory) {
      var yearText = $('<div class="year">').text(image.category);
      $('#gallery').append(yearText);
      currentCategory = image.category;
    }

    var thumbnail = $('<div class="thumbnail">');
    var link = $('<a>').attr('href', image.url).attr('data-fancybox', 'gallery').attr('data-caption', image.title);
    var img = $('<img>').attr('src', image.url);

    img.css({
      top: image.position.top,
      left: image.position.left,
      transform: 'translate(-50%, -50%) scale(' + image.scale + ')'
    });

    img.on('load', function () {
      var aspectRatio = img.width() / img.height();
      if (aspectRatio > 1) {
        img.css('height', '100%');
      } else {
        img.css('width', '100%');
      }
    });

    link.append(img);
    thumbnail.append(link);
    $('#gallery').append(thumbnail);
  });

  $("[data-fancybox='gallery']").fancybox({
    caption: function (instance, item) {
      var captionText = item.opts.caption || '';
      return captionText + '<br><small>' + (item.index + 1) + ' / ' + (item.group ? item.group.length : 1) + '</small>';
    }
  });
